
#check version
docker -v
#to list the images
docker images
#to list the containers
docker container ls -a
#to pull the mongo db image from docker repository
docker pull mongo
#to create mongo db container
docker run --name=niit-mongo-c1 -p 27017:27017 -d mongo
#create mongodb client
docker exec -it <containerid> bin/bash
#mongosh
#to docker build the customer api
docker build -f dockerfile -t customerapi .
#create docker network
docker network create niitnetwork
#create the containers
docker-compose up

#create postgres db
docker exec -it <containerid>  psql -U postgres
#to list the databases
#\l
#to create the database
create database niitdevcustomerdb;
exit

#connect to remote repository
docker login
docker tag <imagename> <dockerhubid/reponame:tagname>
docker push <dockerhubid/reponame:tagname>
